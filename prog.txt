res_div = document.getElementById("sqrt-div");
btn_submit = document.getElementById("submitButton");

btn_submit.addEventListener("click",()=>{
    
    btn_submit.remove();
    res_div.innerHTML += "<p style='color: red'>" + 0 + " = " + false + "</p>" + "<hr>"; 
    for(let i = 1; i<=100; i++)
    {
        let x = Math.sqrt(i);
        let res = (x - Math.floor(x)) == 0;
        
        if(res == true)
        res_div.innerHTML += "<p style='color: green'>" + i + " = " + res + "</p>" + "<hr>";
        else
        res_div.innerHTML += "<p style='color: red'>" + i + " = " + res + "</p>" + "<hr>";
          
    }
    
});
